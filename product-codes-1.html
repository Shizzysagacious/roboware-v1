const int motor1Pin2 = 3;
const int motor2Pin1 = 4;
const int motor2Pin2 = 5;
const int ultrasonicTrigPin = 6;
const int ultrasonicEchoPin = 7;
const int speakerPin = 8;

int motor1Speed = 0;
int motor2Speed = 0;
int ultrasonicDistance = 0;

void setup() {
    pinMode(motor1Pin1, OUTPUT);
    pinMode(motor1Pin2, OUTPUT);
    pinMode(motor2Pin1, OUTPUT);
    pinMode(motor2Pin2, OUTPUT);
    pinMode(ultrasonicTrigPin, OUTPUT);
    pinMode(ultrasonicEchoPin, INPUT);
    pinMode(speakerPin, OUTPUT);

    Serial.begin(9600);
}

void loop() {
    ultrasonicDistance = readUltrasonicDistance();

    if (ultrasonicDistance < 20) {
        playAlarmingSound();
    }

    int signal = readRemoteControllerSignal();

    if (signal == 1) {
        motor1Speed = 255;
        motor2Speed = 255;
    } else if (signal == 2) {
        motor1Speed = -255;
        motor2Speed = -255;
    } else if (signal == 3) {
        motor1Speed = 255;
        motor2Speed = -255;
    } else if (signal == 4) {
        motor1Speed = -255;
        motor2Speed = 255;
    }

    setMotorSpeeds(motor1Speed, motor2Speed);
    delay(50);
}

int readUltrasonicDistance() {
    digitalWrite(ultrasonicTrigPin, HIGH);
    delayMicroseconds(10);
    digitalWrite(ultrasonicTrigPin, LOW);
    int distance = pulseIn(ultrasonicEchoPin, HIGH) * 0.034 / 2;
    return distance;
}

void playAlarmingSound() {
    tone(speakerPin, 1000, 500);
    delay(500);
}

void setMotorSpeeds(int motor1Speed, int motor2Speed) {
    analogWrite(motor1Pin1, motor1Speed > 0 ? motor1Speed : 0);
    analogWrite(motor1Pin2, motor1Speed < 0 ? -motor1Speed : 0);
    analogWrite(motor2Pin1, motor2Speed > 0 ? motor2Speed : 0);
    analogWrite(motor2Pin2, motor2Speed < 0 ? -motor2Speed : 0);
}
